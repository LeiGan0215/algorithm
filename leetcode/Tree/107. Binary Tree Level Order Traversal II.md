class Solution {
public:
    vector<vector<int>> levelOrderBottom(TreeNode* root) {
        vector<vector<int>> res;
        if(!root)
            return res;
        queue<TreeNode*> s;
        s.push(root);
        while(!s.empty())
        {
            int n=s.size();
            vector<int> temp;
            for(int i=0;i<n;i++)
            {
                TreeNode* p=s.front();
                temp.push_back(p->val);
                s.pop();
                if(p->left)
                    s.push(p->left);
                if(p->right)
                    s.push(p->right);
            }
            res.insert(res.begin(), temp);
        }
        return res;
    }
};
